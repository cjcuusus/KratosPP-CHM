显示信息
; 模块：显示信息 ―― AE生效期间，在单位身上显示出监视的AE对象的信息，X=显示的信息，Duration\InitDelay\Delay\Stack
Info.Watch=AE1 ;待监视的AE
Info.X.Watch=AE1 ;微观设置监视的AE
Info.X.Mode=NONE;叠加层数的显示模式，NONE\TEXT\SHP\IMAGE，NONE表示不显示，TEXT=游戏字体，SHP=SHP数字，IMAGE=进度条
Info.X.ShowEnemy=yes ;显示给敌方
Info.X.OnlySelected=no ;只在选择时显示
Info.X.Offset=0,0 ;锚点向右和向下的偏移位置
Info.X.Sort=FIRST;有多个时，显示的排序依据，FIRST\MIN\MAX，FIRST=查找到的第一个，MIN=最小值，MAX=最大值
Info.X.Align=CENTER ;文字的对齐方式，LEFT\CENTER\RIGHT
; 字体设置
Info.X.Color=0,252,0
Info.X.IsHouseColor=no
Info.X.ShadowOffset=1,1
Info.X.ShadowColor=82,85,82
; 使用shp显示数字而不是font显示数字，当模式为Image时，则显示的是shp文件的第(ZeroFrameIndex + 数字 / Wrap)帧对应的图像
Info.X.SHP=pipsnum.shp
Info.X.ZeroFrameIndex=0
Info.X.MaxFrameIndex= -1;模式为Image时，允许显示的帧序号的最大值，如Wrap=1时，MaxFrameIndex=5，则当数字大于5时，仍显示第5帧
Info.X.ImageSize=5,8 ;模式为SHP数字时，每个位数的大小
Info.X.Wrap=1 ;模式为Image时，增加多少个数字跳下一帧
; 监视单位自身的属性，不读取Watch
Info.Health.Mode=NONE ;显示单位的血量
Info.Ammo.Mode=NONE ;显示单位的弹药
Info.Reload.Mode=NONE ;显示单位的弹药装填延迟时间
Info.ROF.Mode=NONE ;显示单位的ROF延迟时间
; 涉及到文字内容或者绘制线条，只能使用特定类型显示的单位属性
Info.ID.Mode=NONE ;使用Text显示出单位的section
Info.Armor.Mode=NONE ;使用Text显示出单位的护甲类型
Info.Mission.Mode=NONE ;使用Text显示出单位的任务
Info.Target.Mode=NONE ;非None时，绘制单位和目标之前的连线
Info.Target.Color=252,0,0 ;连线的颜色
Info.Dest.Mode=NONE ;非None时，绘制单位和移动目标之间的连线
Info.Dest.Color=252,0,0 ;连线的颜色
Info.Location.Mode=NONE ;非None时，使用交叉线绘制目标的位置
Info.Location.Color=0,252,0 ;连线的颜色
Info.Cell.Mode=NONE ;非None时，绘制单位所在的格子的框线
Info.Cell.Color=0,252,0 ;连线的颜色
Info.BodyDir.Mode=NONE ;非None时，绘制单位朝向
Info.BodyDir.Color=0,252,0 ;连线的颜色
Info.TurretDir.Mode=NONE ;非None时，绘制单位炮塔朝向
Info.TurretDir.Color=0,0,252 ;连线的颜色
